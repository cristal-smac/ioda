; [TRIGGER <tr'1> ... <tr'_n>]

; NB1: tr_i and cond_i are abstract condition primitives,

INTERACTION LightAndWalk
  CONDITION can-move?
  ACTIONS   target:become-bright forward
END

INTERACTION TurnRandomly
  TRIGGER   not:can-move?
  ACTIONS   target:become-bright turn-randomly
END

INTERACTION Darken
  CONDITION bright?
  ACTIONS   become-dark
END
